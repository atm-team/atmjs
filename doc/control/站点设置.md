# 站点设置(站点目录下的/atmjs/settings.js文件)
在创建站点时自动生成该文件
```js
var settings = {
    name: 'blog',                      // 站点名称 必填(字母+数字+下划线组成,必须以字母开头)
    libName: 'lib',                    // 公共项目的名称
    // jsonpName: 'callback',
    // timeoutName: 'timeout',
    deploy: {                          // 源文件部署
        option: {                      // 部署选项
            type: 'scp',               // 上传方式

            // 部署到远程服务器的位置,双花括号为占位符,eg:部署的模块的版本为 lib/jquery/1.11.3
            // {{site}} 为settings.name,
            // {{project}} 为 lib
            // {{module}} 为 jquery
            // {{version}} 为 1.11.3
            dist: '/path/to/{{site}}/{{project}}/{{module}}/{{version}}',
            distTemp: '/homedir',           // 远程服务器的homedir目录
            exclusions: {},                 // 排除的文件规则
            globOptions: {},                // glob模块配置
            auth: {                         // 验证信息
                host: 'host',
                username: 'username',
                password: 'password',
                port: 22
            },
            force: false,                   // 当远程目录正在锁定时是否继续上传
            useMap: true                    // 是否差异化上传
        },
        callback: {                         // 上传成功后的回调地址
            url: 'http://hostname/path/to'
        }
    },
    targets: {                              // 目标配置
        dev: {                              // dev目标的配置
            color: 'green',                 // GUI中目标操作按钮的颜色,参考semantic-ui中的按钮颜色的class
            operates: {                     // dev目标可用的操作,指定项必须在operates中存在
                build: 1
            },
            compress: {
                js: false,
                css: false,
                png: false
            }
        },
        test: {                             // test目标的配置
            // 构建后的文件的根路径,可用占位符{{site}}(开发目录,即站点目录)和{{homedir}}(用户目录)
            assets: '{{site}}/../assets/blog-site/test',

            // 产出的地图文件的根路径,可用占位符{{site}}(开发目录,即站点目录)和{{homedir}}(用户目录)
            maps: '{{site}}/../maps/blog-site/test',

            // 静态文件根目录对应的http[s]路径
            domain: '//hostname/path/to',

            color: 'brown',                 // GUI中目标操作按钮的颜色,参考semantic-ui中的按钮颜色的class
            operates: {                     // test目标可用的操作,指定项必须在operates中存在
                build: 1,                   // 构建
                map: 1,                     // 地图
                deploy: 1,                  // 部署
                publish: 1,                 // 发布
                build_map: 1,
                build_map_deploy: 1,
                build_map_deploy_publish: 1
            },
            hash: true,                     // 构建后的文件名是否带7位的md5后缀
            compress: {
                js: true,                   // 构建后的js文件是否进行压缩
                css: true,                  // 构建后的css文件是否进行压缩
                png: true                   // png文件是否进行压缩
            },
            deploy: {                       // 构建后的文件的部署,同settings.deploy.option一致
                option: {
                    type: 'scp',
                    dist: '/path/to/{{site}}/{{project}}/{{module}}/{{version}}',
                    distTemp: '/path/to',
                    exclusions: {},
                    globOptions: {},
                    auth: {
                        host: 'host',
                        username: 'username',
                        password: 'password',
                        port: 22
                    },
                    force: false,
                    useMap: true
                },
                callback: {
                    url: '//hostname/path/to',
                    headers: {}
                }
            },
            publish: {                      // 发布选项,同deploy设置一致
                option: {
                    type: 'scp',
                    dist: '/root/test/publish/{{site}}/{{project}}/{{module}}/{{version}}',
                    distTemp: '/root',
                    exclusions: {},
                    globOptions: {},
                    auth: {
                        host: 'host',
                        username: 'username',
                        password: 'password',
                        port: 22
                    },
                    force: false,
                    useMap: false
                },
                callback: {
                    url: '//hostname/path/to'
                }
            }
        }
    }
};
module.exports = settings;
```
